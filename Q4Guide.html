7) Consider the following program

class Mystery {
  public static int mystery(int[] values) {
    int total = 0;
    for (int i = 0; i < values.length; i++) {
      int value = values[i];
      if (values[i] < 0) {
        value = -value;
      }
      total += value;
      System.out.println("mystery: ", value, total);
    }
    return total;
  }

  public static void main(String[] args) {

void oneLoop(int[] L)
{
   for (int j = 0; j < L.length()-1; j++)
   {
      if (L[j] > L[j+1])
      {
         int tmp = L[j+1];
         L[j+1] = L[j];
         L[j] = tmp;
      }
   }
}
